    /* eslint-disable react-native/no-inline-styles */
import React from 'react';
import {useTranslation} from 'react-i18next';
import {Modal, StyleSheet, Text, TouchableOpacity, View} from 'react-native';
import {
  responsiveFontSize,
  responsiveHeight,
  responsiveWidth,
} from 'react-native-responsive-dimensions';
import {AppColor} from '../constants/Colors';
import BTN from './ButtonLight';
const ModalBox = ({
  visible,
  testData,
  onPressClose,
  onPressConfirm,
  hospitalName,
}) => {
  
  const {t} = useTranslation();
  return (
    <Modal transparent visible={visible}>
      <View style={styles.container}>
        <View style={styles.subContainer1}>
          <Text style={styles.text}>
            {'Would you like to confirm your PCR test result from [' +
              hospitalName +
              '] as '}
            <Text style={{fontWeight: '700'}}>{testData.results + '?'}</Text>
          </Text>
          <View
            style={{
              flexDirection: 'row',
              width: '100%',
              height: responsiveHeight(8),
              // backgroundColor: AppColor.BgColorSecondary,
              borderTopWidth: 0.3,
              borderTopColor: '#7070701A',
            }}>
            <TouchableOpacity onPress={() => onPressClose()} style={styles.btn}>
              <Text
                style={{
                  fontSize: responsiveFontSize(2.4),
                  color: AppColor.TextColorLight,
                }}>
                {t('results.cancel')}
              </Text>
            </TouchableOpacity>
            <TouchableOpacity
              onPress={() => onPressConfirm()}
              style={[
                styles.btn,
                {backgroundColor: AppColor.BgColorSecondary},
              ]}>
              <Text
                style={{
                  fontSize: responsiveFontSize(2.4),
                  color: AppColor.TextColorWhite,
                }}>
                {t('confirmImage.confirm')}
              </Text>
            </TouchableOpacity>
          </View>
        </View>
      </View>
    </Modal>
  );
};

const styles = StyleSheet.create({
  container: {
    flex: 1,
    paddingHorizontal: 15,
    zIndex: 10,
    backgroundColor: 'rgba(0,0,0,0.3)',
    alignItems: 'center',
    justifyContent: 'center',
  },
  subContainer1: {
    justifyContent: 'center',
    alignItems: 'center',
    width: responsiveWidth(85),
    backgroundColor: AppColor.BgColorPrimary,
    overflow: 'hidden',
    borderRadius: responsiveWidth(5),
  },
  text: {
    fontSize: responsiveFontSize(2.5),
    width: responsiveWidth(80),
    textAlign: 'center',
    margin: responsiveHeight(3),
  },
  btn: {
    width: '50%',
    height: responsiveHeight(8.5),
    alignItems: 'center',
    justifyContent: 'center',
    backgroundColor: AppColor.BgColorPrimary,
  },
});

export default ModalBox;

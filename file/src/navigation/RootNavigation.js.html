    import {NavigationContainer} from '@react-navigation/native';
import SplashScreen from 'react-native-splash-screen';
import React, {useEffect} from 'react';
import {
  MY_INFO,
  MY_PASS,
  MY_PROFILE,
  SETTINGS,
  SITE_MANAGER,
} from '../constants/Routes';
import Auth from './Auth';
import AuthenticatedTabNavigator, {
  CommonStack,
} from './AuthenticatedTabNavigator';
import {useSelector} from 'react-redux';
import {Text} from 'react-native';
import {getData} from '../utils/asyncRequests';
import {TOKEN, USER_DATA} from '../constants/Keys';
import {useDispatch} from 'react-redux';
import {setCurrentUser, setToken} from '../redux/actions/user';
import {navigationRef} from './RootNavigationWithoutProp';
import {createStackNavigator} from '@react-navigation/stack';
import MyInfo from '../screens/authenticated/myInfo';
import {SettingsStack} from './AuthenticatedTabNavigator';
import ChangePassword from '../screens/authenticated/changePassword';
import LoadingSplash from '../screens/authenticated/LoadingSplash';
import Success2 from '../screens/authenticated/success';
import PassService from '../screens/authenticated/paasService';
import ResetPassword from '../screens/authenticated/resetPassword';

const Stack = createStackNavigator();

const config = {
  screens: {
    // [MY_PASS]: {
    //   screens: {
    //     [MY_INFO]: 'my-info',
    //   },
    //   // parse: {
    //   //   data: data => `${data}`,
    //   // },
    // },
    [MY_INFO]: {
      path: 'my-info/:data',

      parse: {
        data: data => `${data}`,
      },
    },

    [SITE_MANAGER]: {
      path: 'site-manager',
    },

    [MY_PROFILE]: {
      path: 'my-profile/:data',
      parse: {
        data: data => `${data}`,
      },
    },
  },
};

const linking = {
  prefixes: ['passngo://app'],
  config,
};

export default () => {
  const token = useSelector(state => state.user.token);
  const loading = useSelector(state => state.user.loading);
  const dispatch = useDispatch();
  useEffect(() => {
    setTimeout(() => {
      SplashScreen.hide();
    }, 1000);
  }, []);
  useEffect(() => {
    /**
     * Get current Language
     * @param {string} lang
     */
    getData(USER_DATA).then(data => {
      if (data != null) {
        dispatch(setCurrentUser(JSON.parse(data)));
      }
    });
    getData(TOKEN).then(data => {
      if (data != null) {
        dispatch(setToken(JSON.parse(data)));
      }
    });
  });
  return (
    <NavigationContainer
      // fallback={<Text>Loding...</Text>}
      linking={linking}
      ref={navigationRef}>
      {/* {!token || token.length === 0 ? <Auth /> : <AuthenticatedTabNavigator />} */}
      <Stack.Navigator screenOptions={{headerShown: false}}>
        {!loading ? (
          <Stack.Screen
            name={'auth'}
            component={
              !token || token.length === 0 ? Auth : AuthenticatedTabNavigator
            }
          />
        ) : (
          <Stack.Screen name={'load_screen'} component={LoadingSplash} />
        )}
        <Stack.Screen
          name={MY_INFO}
          component={!token || token.length === 0 ? Auth : MyInfo}
        />
        <Stack.Screen
          name={'passserve'}
          component={!token || token.length === 0 ? Auth : PassService}
        />
        <Stack.Screen
          name={'successpass'}
          component={!token || token.length === 0 ? Auth : Success2}
        />
        <Stack.Screen
          name={'resetpass'}
          component={!token || token.length === 0 ? Auth : ResetPassword}
        />
        <Stack.Screen name={'COMMON'} component={CommonStack} />
        <Stack.Screen name={'change_password'} component={ChangePassword} />
      </Stack.Navigator>
    </NavigationContainer>
  );
};

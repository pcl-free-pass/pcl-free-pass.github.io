    src/screens/auth/signUpScreen/index.js

    import React, {useState} from 'react';
import {
  SafeAreaView,
  StatusBar,
  StyleSheet,
  Text,
  View,
  TextInput,
  TouchableOpacity,
  Platform,
  ScrollView,
} from 'react-native';
import {useTranslation} from 'react-i18next';
import {
  responsiveWidth,
  responsiveHeight,
  responsiveFontSize,
} from 'react-native-responsive-dimensions';
import DateTimePicker from '@react-native-community/datetimepicker';
import {Picker} from '@react-native-picker/picker';
import Button from '../../../components/Button';
import Separator from '../../../components/Separator';
import {AppColor} from '../../../constants/Colors';
import {SIGN_IN, SET_PASSWORD} from '../../../constants/Routes';
import Logo from '../../../assets/images/logos/PnGFull.svg';
import Snackbar from 'react-native-snackbar';
import moment from 'moment';

const SignUpScreen = ({navigation}) => {
  const [name, setName] = useState('');
  const [number, setNumber] = useState('');
  const [id, setId] = useState('');
  const [dob, setDob] = useState('');
  const [show, setShow] = useState(false);
  const [gender, setGender] = useState('Male');
  const {t, i18n} = useTranslation('en');

  /**
   * validate inputs and then go to next screen
   */
  const validateInput = () => {
    let pattern = '\\d{1}\\d{10}';
    if (!number.match(pattern) && number) {
      Snackbar.show({
        text: t('common.enterNumber'),
        backgroundColor: AppColor.BgColorSecondary,
        textColor: AppColor.PrimaryLight,
        duration: Snackbar.LENGTH_SHORT,
      });
      return;
    }
    let patternId = '\\d{13}';
    if (!id.match(patternId) && id) {
      Snackbar.show({
        text:t('common.validId'),
        backgroundColor: AppColor.BgColorSecondary,
        textColor: AppColor.PrimaryLight,
        duration: Snackbar.LENGTH_SHORT,
      });
      return;
    }
    if (name && number && id && dob && gender) {
      let obj = {
        name: name,
        password: '',
        phone: number,
        dob: 20202020,
        photo: '',
        gender: 1,
      };
      navigation.navigate(SET_PASSWORD, {obj: obj});
    } else {
      Snackbar.show({
        text: t('common.fillAll'),
        textColor: AppColor.TextColorWhite,
        backgroundColor: AppColor.BgColorSecondary,
        duration: Snackbar.LENGTH_SHORT,
      });
    }
  };
  /**
   * Check number for validation.
   */
  const checkNumber = async () => {
    let pattern = '\\d{2}\\d{10}';
    if (!number.match(pattern) && number) {
      Snackbar.show({
        text: t('common.enterNumber'),
        backgroundColor: AppColor.BgColorSecondary,
        textColor: AppColor.PrimaryLight,
        duration: Snackbar.LENGTH_SHORT,
      });
      return;
    }
  };
  return (
    <SafeAreaView style={styles.safeAreaView}>
      <View style={styles.container}>
        {/* <StatusBar translucent hidden={true} /> */}
        <ScrollView
          contentContainerStyle={styles.scrollView}
          showsVerticalScrollIndicator={false}>
          <StatusBar
            barStyle={'dark-content'}
            backgroundColor={AppColor.Secondary}
          />
          <Logo
            height={responsiveHeight(30)}
            width={responsiveWidth(65)}
            fill={AppColor.BgColorSecondary}
          />
          <View style={styles.inputView}>
            <TextInput
              style={styles.input}
              placeholder={'Name'}
              onChangeText={val => setName(val)}
              placeholderTextColor={AppColor.TextColorLight}
            />
          </View>
          <View style={styles.inputView}>
            <TextInput
              keyboardType={'phone-pad'}
              style={styles.input}
              maxLength={12}
              placeholder={'Phone number(923125590456)'}
              onChangeText={val => setNumber(val)}
              placeholderTextColor={AppColor.TextColorLight}
              onBlur={() => checkNumber()}
            />
          </View>
          <View style={styles.inputView}>
            <TextInput
              style={styles.input}
              keyboardType={'decimal-pad'}
              placeholder={'ID card'}
              maxLength={13}
              onChangeText={val => {
                setId(val);
              }}
              placeholderTextColor={AppColor.TextColorLight}
            />
          </View>
          <TouchableOpacity
            onPress={() => setShow(!show)}
            style={[styles.inputView]}>
            <Text
              style={[
                styles.input,
                {
                  color: dob ? AppColor.TextColor : AppColor.TextColorLight,
                  textAlignVertical: 'center',
                  paddingTop: 17,
                  height: '100%',
                },
              ]}>
              {dob ? dob : 'Date of Birth'}
            </Text>
          </TouchableOpacity>
          {Platform.OS === 'android' && (
            <View style={styles.inputView}>
              <Picker
                selectedValue={gender}
                style={styles.input}
                mode={'dialog'}
                onValueChange={(itemValue, itemIndex) => setGender(itemValue)}>
                <Picker.Item label="Male" value="Male" />
                <Picker.Item label="Female" value="Female" />
              </Picker>
            </View>
          )}
          {Platform.OS === 'ios' && (
            <View>
              <Picker
                selectedValue={gender}
                style={styles.input2}
                mode={'dialog'}
                onValueChange={(itemValue, itemIndex) => setGender(itemValue)}>
                <Picker.Item label="Male" value="Male" />
                <Picker.Item label="Female" value="Female" />
              </Picker>
              <Separator height={responsiveHeight(15)} />
            </View>
          )}
          <Separator height={responsiveHeight(5)} />
          <Button onPress={() => validateInput()} title={'Sign Up'} />
          <Separator height={responsiveHeight(2)} />
          <View style={{flexDirection: 'row'}}>
            <Text style={styles.dontHav}>{'Already you have account?'}</Text>
            <TouchableOpacity onPress={() => navigation.navigate(SIGN_IN)}>
              <Text style={styles.forgot}>{'  Sign In'}</Text>
            </TouchableOpacity>
          </View>
          <Separator height={responsiveHeight(6)} />
        </ScrollView>
        {show && (
          <DateTimePicker
            value={new Date()}
            style={{textColor: 'red', width: '100%', alignSelf: 'center'}}
            onChange={val => {
              let newdate = moment(val.nativeEvent.timestamp).format(
                'YYYY-MM-DD',
              );
              setShow(!show);
              setDob(newdate);
            }}
          />
        )}
      </View>
    </SafeAreaView>
  );
};

const styles = StyleSheet.create({
  safeAreaView: {
    flex: 1,
    backgroundColor: '#fff',
  },
  container: {
    flex: 1,
    backgroundColor: '#fff',
    alignItems: 'center',
    justifyContent: 'center',
  },
  scrollView: {
    backgroundColor: '#fff',
    alignItems: 'center',
  },
  image: {
    width: responsiveWidth(90),
    height: responsiveHeight(15),
    resizeMode: 'contain',
  },
  inputView: {
    height: responsiveHeight(7),
    borderRadius: responsiveHeight(7),
    borderWidth: 1.5,
    borderColor: AppColor.BgColorSecondary,
    marginVertical: responsiveHeight(1),
    justifyContent: 'center',
    width: responsiveWidth(85),
    alignItems: 'center',
    flexDirection: 'row',
  },
  input: {
    width: responsiveWidth(75),
    alignSelf: 'center',
    height: responsiveHeight(5),
    padding: 0,
    color: AppColor.TextColor,

    fontSize: responsiveFontSize(2.1),
  },
  input2: {
    width: responsiveWidth(80),
    alignSelf: 'center',
    padding: 0,
    color: AppColor.TextColor,
    fontSize: responsiveFontSize(2.1),
    height: 0,
  },
  forgot: {
    fontSize: responsiveFontSize(2),
    color: AppColor.TextColorSecondary,
  },
  dontHav: {
    fontSize: responsiveFontSize(2),
    color: AppColor.TextColor,
  },
});

export default SignUpScreen;

    import React, {useEffect, useState} from 'react';
import {
  SafeAreaView,
  StyleSheet,
  Text,
  View,
  TouchableOpacity,
} from 'react-native';
import {useTranslation} from 'react-i18next';
import {
  responsiveWidth,
  responsiveHeight,
  responsiveFontSize,
} from 'react-native-responsive-dimensions';
import {AppColor} from '../../../constants/Colors';
import Logo from '../../../assets/images/instructionsResults/success.svg';
import Separator from '../../../components/Separator';
import Button from '../../../components/Button';
import Headerlight from '../../../components/Headerlight';
import {getData, saveData} from '../../../utils/asyncRequests';
import {APP_LANG} from '../../../constants/Keys';

/**
 * this functional component is for using change language
 * @param {object} param0 navigation and route props
 * @returns component
 */
const ChangeLanguage = ({navigation}) => {
  const [selectedLanguage, setSelectedLanguage] = useState('en');
  const {t, i18n} = useTranslation('en');

  useEffect(() => {
    /**
     * Get currently selected language
     */
    getData(APP_LANG).then(data => {
      if (data != null) {
        setSelectedLanguage(data);
      } else {
        setSelectedLanguage('en');
      }
    });
  }, []);

  /**
   * Change current language
   * @param {string} lang selected language
   */
  const setLanguage = lang => {
    setSelectedLanguage(lang);
  };

  /**
   * Confirm Change language
   * @param {string} lang selected language
   */
  const confirmLanguage = lang => {
    i18n.changeLanguage(lang);
    saveData(APP_LANG, lang);
    navigation.goBack(null);
  };

  return (
    <SafeAreaView style={styles.safeAreaView}>
      <View style={styles.container}>
        <Headerlight title={t('about.change')} navigation={navigation} />
        <TouchableOpacity
          onPress={() => setLanguage('en')}
          style={styles.rowMainView}>
          <Text style={styles.language}>English</Text>
          <View style={styles.rowView}>
            {selectedLanguage == 'en' && (
              <Logo
                height={responsiveWidth(9)}
                width={responsiveWidth(9)}
                fill={AppColor.TextColorSecondary}
              />
            )}
          </View>
        </TouchableOpacity>
        <TouchableOpacity
          onPress={() => setLanguage('ko')}
          style={styles.rowMainView}>
          <Text style={styles.language}>한국어</Text>
          <View style={styles.rowView}>
            {selectedLanguage == 'ko' && (
              <Logo
                height={responsiveWidth(9)}
                width={responsiveWidth(9)}
                fill={AppColor.TextColorSecondary}
              />
            )}
          </View>
        </TouchableOpacity>
        <TouchableOpacity
          onPress={() => setLanguage('zn')}
          style={styles.rowMainView}>
          <Text style={styles.language}>中文</Text>
          <View style={styles.rowView}>
            {selectedLanguage == 'zn' && (
              <Logo
                height={responsiveWidth(9)}
                width={responsiveWidth(9)}
                fill={AppColor.TextColorSecondary}
              />
            )}
          </View>
        </TouchableOpacity>
        <TouchableOpacity
          onPress={() => setLanguage('ja')}
          style={styles.rowMainView}>
          <Text style={styles.language}>日本語</Text>
          <View style={styles.rowView}>
            {selectedLanguage == 'ja' && (
              <Logo
                height={responsiveWidth(9)}
                width={responsiveWidth(9)}
                fill={AppColor.TextColorSecondary}
              />
            )}
          </View>
        </TouchableOpacity>
        <Separator height={responsiveHeight(30)} />
        <Button
          onPress={() => {
            confirmLanguage(selectedLanguage);
          }}
          title={t('confirmImage.confirm')}
          width={responsiveWidth(80)}
        />
      </View>
    </SafeAreaView>
  );
};

const styles = StyleSheet.create({
  safeAreaView: {
    flex: 1,
    backgroundColor: AppColor.BgColorPrimary,
  },
  container: {
    flex: 1,
    backgroundColor: AppColor.BgColorPrimary,
    alignItems: 'center',
    justifyContent: 'center',
  },
  language: {
    color: AppColor.TextColorDark,
    fontSize: responsiveFontSize(2.2),
  },
  rowMainView: {
    width: responsiveWidth(80),
    height: responsiveHeight(9),
    borderBottomWidth: 1,
    borderBottomColor: '#6D6D6D80',
    alignItems: 'center',
    justifyContent: 'space-between',
    flexDirection: 'row',
  },
  rowView: {
    width: responsiveWidth(9),
    height: responsiveWidth(9),
  },
});

export default ChangeLanguage;

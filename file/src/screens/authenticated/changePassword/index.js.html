    import React, {useEffect, useState} from 'react';
import {
  SafeAreaView,
  StatusBar,
  StyleSheet,
  Text,
  View,
  TextInput,
} from 'react-native';
import {useTranslation} from 'react-i18next';
import AsyncStorage from '@react-native-async-storage/async-storage';
import {
  responsiveWidth,
  responsiveHeight,
  responsiveFontSize,
} from 'react-native-responsive-dimensions';
import {APP_LANG} from '../../../constants/Keys';
import Button from '../../../components/Button';
import Separator from '../../../components/Separator';
import {AppColor} from '../../../constants/Colors';
import Logo from '../../../assets/images/logos/PnGSmall.svg';
import Headerlight from '../../../components/Headerlight';
import {CHANGE_PASSWORD_URL} from '../../../constants/ApiRoutes';
import {useSelector} from 'react-redux';
import Snackbar from 'react-native-snackbar';
import {PostRequest} from '../../../utils/apiRequests';
import {
  COMMON_ROUTES,
  FORGOT_PASSWORD,
  MY_PASS,
  PAAS_SERVICE,
  PASS_SERVICE,
  SUCCESS,
} from '../../../constants/Routes';
import {KeyboardAwareScrollView} from 'react-native-keyboard-aware-scrollview';

/**
 * this functional component is for changing password
 * @param {object} param0 navigation and route props
 * @returns component
 */
const ChangePassword = ({navigation}) => {
  const {t, i18n} = useTranslation('en');
  const [currentPassword, setcurrentPassword] = useState('');
  const token = useSelector(state => state.user.token);
  const [newPassword, setPassword] = useState('');
  const [confirmPassword, setConfirmPassword] = useState('');
  const [validated, setvalidated] = useState();
  const [processing, setProcessing] = useState(false);

  /**
   * Request to change current password
   * @returns Promise
   */
  const changePassword = async () => {
    if (currentPassword.trim().length === 0) {
      showSnackBar(t('changePassword.confirmPass'));
      return;
    }
    if (newPassword.trim().length === 0) {
      showSnackBar(t('changePassword.newPass'));
      return;
    }
    if (confirmPassword.trim().length === 0) {
      showSnackBar(t('common.enterConfirmPass'));
      return;
    }
    if (!validated) {
      setProcessing(true);
      const data = {
        newPassword,
        currentPassword,
      };
      await PostRequest(CHANGE_PASSWORD_URL, data, 'Token ' + token)
        .then(response => {
          if (response.status === 200) {
            showSnackBar(t('changePassword.changeSuccess'));
            navigation.navigate(MY_PASS);
          }
        })
        .catch(error => {
          if (error.response.status === 400) {
            if (
              error.response.data.message ===
              'Your phone number is not registered'
            ) {
              showSnackBar(t('common.numberNotRegister'));
            }
            if (error.response.data.message === 'Current password is invalid') {
              showSnackBar(t('changePassword.invalidPass'));
            }
            if (
              error.response.data.message ===
              'Current and new passwords are same'
            ) {
              showSnackBar(t('changePassword.passowrdsRSame'));
            }
          } else {
            showSnackBar(t('changePassword.unknownError'));
          }
        })
        .finally(() => {
          setProcessing(false);
        });
    } else {
      showSnackBar(validated);
    }
  };

  /**
   * Validate input passwords
   * @param {string} pass Password
   */
  const CheckPassword = pass => {
    //^(?=.*[!@#$%^&*])(?=.*[a-z])(?=.*[A-Z]).{6,20}$/;
    var passw = /^(?=.*?[A-Z])(?=.*?[a-z])(?=.*?[#?!@$%^&*-]).{8,}$/;
    if (!pass.match(passw) && pass.length > 0) {
      setvalidated(t('common.invalidFormat'));
      showSnackBar(t('common.invalidFormat'));
    } else {
      if (newPassword && confirmPassword && currentPassword) {
        if (newPassword === confirmPassword) {
          setvalidated(null);
        } else {
          setvalidated(t('common.passNotSame'));
          showSnackBar(t('common.passNotSame'));
        }
      } else {
        setvalidated('Fill all the fields');
      }
    }
  };

  /**
   * This method generalize using of Snackbar
   * @param {string} message message to show on snackbar
   */
  const showSnackBar = message => {
    Snackbar.show({
      text: message,
      backgroundColor: AppColor.BgColorSecondary,
      textColor: AppColor.PrimaryLight,
      duration: Snackbar.LENGTH_SHORT,
    });
  };

  /**
   * Action to perform when forget password is pressed
   */
  const onPressForgetPassword = () => {
    // navigation.navigate(SUCCESS, {
    //   screen: 'forgot',
    // });
    navigation.navigate(COMMON_ROUTES, {
      screen: PASS_SERVICE,
      params: {
        type: 'forget_password',
      },
    });
  };

  return (
    <SafeAreaView style={styles.safeAreaView}>
      <View style={styles.container}>
        <Headerlight title={''} navigation={navigation} />

        <View
          style={{position: 'absolute', top: responsiveHeight(0), zIndex: 2}}>
          <Logo
            height={responsiveHeight(10)}
            width={responsiveHeight(10)}
            fill={AppColor.BgColorSecondary}
          />
        </View>
        <KeyboardAwareScrollView
          showsVerticalScrollIndicator={false}
          contentContainerStyle={styles.scrollView}>
          <Separator height={responsiveHeight(18)} />
          <Text style={styles.changeText}>{t('changePassword.title')}</Text>
          <Separator height={responsiveHeight(8)} />

          <View style={styles.inputView}>
            <TextInput
              style={styles.input}
              secureTextEntry={true}
              placeholder={t('changePassword.previousPass')}
              placeholderTextColor={AppColor.TextColorLight}
              onChangeText={val => {
                setcurrentPassword(val);
              }}
              onBlur={() => CheckPassword(currentPassword)}
            />
          </View>
          <Text
            style={styles.forgotText}
            onPress={() => onPressForgetPassword()}>
            {t('changePassword.forgot')}
          </Text>
          <View style={styles.inputView}>
            <TextInput
              style={styles.input}
              secureTextEntry={true}
              placeholder={t('changePassword.newPass')}
              placeholderTextColor={AppColor.TextColorLight}
              onChangeText={val => {
                setPassword(val);
              }}
              onBlur={() => CheckPassword(newPassword)}
            />
          </View>
          <Text style={styles.terms}>{t('changePassword.terms')}</Text>
          <View style={styles.inputView}>
            <TextInput
              style={styles.input}
              secureTextEntry={true}
              placeholder={t('changePassword.confirmPass')}
              placeholderTextColor={AppColor.TextColorLight}
              onChangeText={val => {
                setConfirmPassword(val);
              }}
              onBlur={() => CheckPassword(confirmPassword)}
            />
          </View>
          <Separator height={responsiveHeight(3)} />
          <Button
            showLoading={processing}
            onPress={() => changePassword()}
            title={t('setPassword.next')}
          />
          <Separator height={responsiveHeight(4)} />
        </KeyboardAwareScrollView>
      </View>
    </SafeAreaView>
  );
};

const styles = StyleSheet.create({
  safeAreaView: {
    flex: 1,
    backgroundColor: '#fff',
  },
  container: {

  },
  inputView: {

    marginVertical: responsiveHeight(1),
    justifyContent: 'center',
    width: responsiveWidth(85),
    alignItems: 'center',
    flexDirection: 'row',
  },
  input: {
    width: responsiveWidth(75),
    alignSelf: 'center',
    height: responsiveHeight(5),
    padding: 0,
    color: AppColor.TextColor,
    fontSize: responsiveFontSize(2.1),
  },
  changeText: {
    fontSize: responsiveFontSize(2.5),

    textAlign: 'center',
  },
  forgotText: {
    fontSize: responsiveFontSize(2.2),
    marginVertical: responsiveHeight(1.3),
    color: AppColor.TextColorSecondary,
    width: responsiveWidth(75),

  },
  note: {
    fontSize: responsiveFontSize(2),

  terms: {

    marginVertical: responsiveHeight(1),
    color: AppColor.TextColorSecondary,
    width: responsiveWidth(75),
  },
});

export default ChangePassword;

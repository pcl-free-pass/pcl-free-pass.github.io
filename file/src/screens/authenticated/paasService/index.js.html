    import React from 'react';
import {useTranslation} from 'react-i18next';
import {
  ActivityIndicator,
  SafeAreaView,
  StatusBar,
  StyleSheet,
  TextInput,
  View,
} from 'react-native';
import {
  responsiveWidth,
  responsiveHeight,
} from 'react-native-responsive-dimensions';
import Snackbar from 'react-native-snackbar';
import {WebView} from 'react-native-webview';
import {useDispatch} from 'react-redux';
import {AppColor} from '../../../constants/Colors';
import {TOKEN, USER_DATA} from '../../../constants/Keys';
import {RESET_PASSWORD, SET_PASSWORD} from '../../../constants/Routes';
import {setCurrentUser, setToken} from '../../../redux/actions/user';
import {saveData} from '../../../utils/asyncRequests';

const SignUp = ({navigation, route}) => {
  const [currentUrl, setCurrentUrl] = React.useState(
    'http://52.78.217.251:8888/passview',
  );
  const [loading, setLoading] = React.useState(true);
  const [hardware, setHardware] = React.useState(false);
  const routeData = route.params;
  const dispatch = useDispatch();
  const {t} = useTranslation('en');

  React.useEffect(() => {
    const unsubscribe = navigation.addListener('transitionStart', ({data}) => {
      if (data.closing) {
        setHardware(true);
      }
    });

    return unsubscribe;
  }, [navigation]);

  React.useEffect(() => {
    const unsubscribe = navigation.addListener('transitionEnd', ({data}) => {
      if (!data.closing) {
        setHardware(false);
      }
    });

    return unsubscribe;
  }, [navigation]);

  const _bridge = async event => {
    if (event.nativeEvent.data) {
      const data = JSON.parse(event.nativeEvent.data); //JSON.parse(event.nativeEvent.data);
      if (data.status === 'success') {
        let obj = {
          name: data.data.name,
          phone: data.data.number,
          dob: data.data.dob,
          gender: data.data.gender,
        };

        if (routeData.type === 'signup') {
          navigation.navigate(SET_PASSWORD, {obj: obj});
        } else if (routeData.type === 'signin_verification') {
          showSnackBar(t('login.loginSuccess'));
          dispatch(setToken(routeData.token));
          await saveData(TOKEN, JSON.stringify(routeData.token));

          dispatch(setCurrentUser(routeData.currentUser));
          await saveData(USER_DATA, JSON.stringify(routeData.currentUser));
          await saveData(obj.phone, JSON.stringify(obj));
        } else if (routeData.type === 'forget_password') {
          navigation.navigate('resetpass', {phone: obj.phone});
        }
      } else {
        showSnackBar(t('common.serviceNotAvail'));
        navigation.goBack();
      }
    }
  };

  const _onNavigationStateChange = webViewState => {
    setCurrentUrl(webViewState.url);
  };

  const _onHideLoader = value => {
    setLoading(value);
  };

  const showSnackBar = message => {
    Snackbar.show({
      text: message,
      backgroundColor: AppColor.BgColorSecondary,
      textColor: AppColor.PrimaryLight,
      duration: Snackbar.LENGTH_SHORT,
    });
  };

  return (
    <SafeAreaView style={styles.safeAreaView}>
      <View style={styles.container}>
        <StatusBar
          barStyle={'dark-content'}
          // backgroundColor={AppColor.Secondary}
        />
        <View style={styles.inputView}>
          <View style={styles.inputViewContainer}>
            <TextInput
              style={styles.input}
              value={currentUrl}
              editable={false}
            />
            {loading && (
              <ActivityIndicator
                color={AppColor.Secondary}
                size={'small'}
                // style={{width: '20%', height: '100%'}}
              />
            )}
          </View>
        </View>
        <View style={styles.webviewContainer}>
          <WebView
            style={styles.webview}
            source={{uri: 'http://52.78.217.251:8888/passview'}}
            onError={() => console.log('error')}
            onMessage={event => _bridge(event)}
            javaScriptEnabled={true}
            onNavigationStateChange={_onNavigationStateChange}
            onLoad={() => _onHideLoader(false)}
            onLoadStart={() => _onHideLoader(true)}
            // renderToHardwareTextureAndroid={hardware}
          />
        </View>
      </View>
    </SafeAreaView>
  );
};

const styles = StyleSheet.create({
  safeAreaView: {
    flex: 1,
    backgroundColor: '#fff',
  },
  container: {
    flex: 1,
    backgroundColor: '#fff',
    alignItems: 'center',
    justifyContent: 'center',
  },
  inputViewContainer: {
    height: '90%',
    backgroundColor: AppColor.PrimaryLight,
    borderRadius: 10,
    flexDirection: 'row',
    justifyContent: 'space-between',
    paddingHorizontal: 15,
    alignItems: 'center',
  },
  inputView: {
    height: responsiveHeight(8),
    width: '100%',
    backgroundColor: AppColor.Secondary,
    paddingHorizontal: 15,
    paddingVertical: 10,
  },
  input: {
    height: '100%',
    width: '80%',
    backgroundColor: AppColor.PrimaryLight,
    borderRadius: 10,
    color: AppColor.TextColorLight,
    textAlignVertical: 'center',
  },
  webviewContainer: {
    height: responsiveHeight(86),
    width: responsiveWidth(100),
  },
  webview: {
    flex: 1,
    opacity: 0.99,
  },
});

export default SignUp;

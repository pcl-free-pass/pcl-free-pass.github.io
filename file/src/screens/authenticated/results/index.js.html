/**
 * Results type selection
 */
import React, {useEffect, useState} from 'react';
import {
  SafeAreaView,
  StatusBar,
  StyleSheet,
  Text,
  View,
  TouchableOpacity,
  Platform,
  Dimensions,
  ScrollView,
} from 'react-native';
import {useTranslation} from 'react-i18next';
import {
  responsiveWidth,
  responsiveHeight,
  responsiveFontSize,
} from 'react-native-responsive-dimensions';
import {AppColor} from '../../../constants/Colors';
import Header from '../../../components/Header';
import Qr from '../../../assets/images/util/qr.svg';
import {
  IMPORT_TEST,
  IMPORT_VACCINE,
  RESULTS_ROUTES,
  ST_QR_CODE_SCAN,
} from '../../../constants/Routes';
import ModalBox from '../../../components/ModalBox';
import {useSelector} from 'react-redux';
import QRScanner from '../../../components/QRScanner';
import ResultModal from '../../../components/ResultModal';
import ActionButton from 'react-native-action-button';

var height = Dimensions.get('screen').height;

/**
 * this functional component is to show results screen.
 * so user can select action.
 * @param {object} param0 navigation and route props
 * @returns component
 */
const Results = ({navigation}) => {
  const [modalVisible, setModalVisible] = useState(false);
  const [qrData, setQrData] = useState(JSON.stringify({createdAt: new Date()}));
  const {t, i18n} = useTranslation('en');
  const [qrScanner, setQrScanner] = useState(false);
  const currentUser = useSelector(state => state.user.currentUser);
  const [message, setMessage] = useState(false);
  const [resModalVisible, setResModalVisible] = useState(false);
  const lang = i18n.language;

  useEffect(() => {
    setInterval(() => {
      let data = {
        uuid: currentUser.uuid,
        createdAt: new Date(),
        pcrTestResults: '',
        vaccination: '',
      };

      setQrData(JSON.stringify(data));
    }, 15000);
  }, [currentUser]);

  /**
   * Method to close modal
   */
  const onPressClose = () => {
    setModalVisible(false);
  };

  /**
   * Method to show results
   */
  const onCloseResults = () => {
    setResModalVisible(false);
  };

  /**
   * Method to close QRScanner and to decide next action based on qrResults
   * @param {string} qrResults qrResult message
   * @param {string} site site name
   */
  const onPressCloseScanner = (qrResults, site) => {
    setQrScanner(false);
    if (qrResults) {
      if (qrResults === 'matched') {
        setMessage(t('myPass.welcome', {siteName: site}));
        setResModalVisible(true);
      } else if (qrResults === 'oldpass_valid') {
        setMessage(t('myPass.validOldPass'));
        setResModalVisible(true);
      } else {
        setMessage(t('myPass.informationNotMatching'));
        setResModalVisible(true);
      }
    }
  };

  return (
    <SafeAreaView style={styles.safeAreaView}>
      <StatusBar
        barStyle={'light-content'}
        backgroundColor={AppColor.Secondary}
      />
      <QRScanner
        visible={qrScanner}
        userType={'user'}
        onPressCloseScanner={onPressCloseScanner}
      />
      {!qrScanner ? (
        <View style={styles.container}>
          <ModalBox visible={modalVisible} onPressClose={onPressClose} />

          <ResultModal
            visible={resModalVisible}
            message={message}
            onPressClose={onCloseResults}
          />
          <Header title={t('results.title')} navigation={navigation} />
          <View style={styles.upperView}>
            <View
              style={{
                ...Platform.select({
                  android: {
                    marginTop:
                      height < 800
                        ? responsiveHeight(10)
                        : responsiveHeight(16),
                  },
                  ios: {
                    marginTop:
                      height < 800
                        ? responsiveHeight(10)
                        : responsiveHeight(12),
                  },
                }),
              }}>
              <ScrollView
                showsVerticalScrollIndicator={false}
                contentContainerStyle={{paddingBottom: 140}}>
                <TouchableOpacity
                  onPress={() =>
                    navigation.navigate(RESULTS_ROUTES, {
                      screen: IMPORT_TEST,
                    })
                  }
                  style={styles.QrView}>
                  <Text style={styles.heading}>{t('results.impotTest')}</Text>
                  <Text style={styles.text}>
                    {Platform.OS === 'ios'
                      ? t('results.pcrNote')
                      : t('results.pcrNoteAndroid')}
                  </Text>
                </TouchableOpacity>

                <TouchableOpacity
                  onPress={() =>
                    navigation.navigate(RESULTS_ROUTES, {
                      screen: ST_QR_CODE_SCAN,
                    })
                  }
                  style={[
                    styles.QrView,
                    {marginTop: responsiveHeight(2), borderColor: '#FF3A3A'},
                  ]}>
                  <Text style={styles.heading}>
                    {t('results.importSelfTest')}
                  </Text>
                  <Text style={styles.text}>
                    {Platform.OS === 'ios'
                      ? t('results.selfTestNote')
                      : t('results.selfTestNoteAndroid')}
                  </Text>
                </TouchableOpacity>

                <TouchableOpacity
                  onPress={() =>
                    navigation.navigate(RESULTS_ROUTES, {
                      screen: IMPORT_VACCINE,
                    })
                  }
                  style={[
                    styles.QrView,
                    {marginTop: responsiveHeight(2), borderColor: '#00E6BF'},
                  ]}>
                  <Text style={styles.heading}>
                    {t('results.importVaccine')}
                  </Text>
                  <Text style={styles.text}>
                    {t('results.uploadCertificate')}
                  </Text>
                </TouchableOpacity>
              </ScrollView>
            </View>
            <ActionButton
              buttonColor={AppColor.FloatingActionButton}
              size={responsiveFontSize(7)}
              offsetX={0}
              offsetY={6}
              onPress={() => setQrScanner(true)}
              renderIcon={() => (
                <Qr
                  height={responsiveFontSize(4)}
                  width={responsiveFontSize(4)}
                  fill={AppColor.PrimaryLight}
                />
              )}
            />
          </View>

          <View style={styles.middleView} />
        </View>
      ) : (
        <View style={styles.emptyView} />
      )}
    </SafeAreaView>
  );
};

    /* eslint-disable react-native/no-inline-styles */
/* eslint-disable no-shadow */
import React, {useEffect, useState} from 'react';
import {
  SafeAreaView,
  StyleSheet,
  Image,
  Text,
  View,
  TouchableOpacity,
  TextInput,
} from 'react-native';
import {useTranslation} from 'react-i18next';
import {
  responsiveWidth,
  responsiveHeight,
  responsiveFontSize,
} from 'react-native-responsive-dimensions';
import Button from '../../../../components/Button';
import Separator from '../../../../components/Separator';
import {AppColor} from '../../../../constants/Colors';
import Headerlight from '../../../../components/Headerlight';
import Cal from '../../../../assets/images/timeSetup/cal.svg';
import DropDown from '../../../../assets/images/util/dropDown.svg';
import Snackbar from 'react-native-snackbar';
import {useDispatch, useSelector} from 'react-redux';
import {SUCCESS_TEST} from '../../../../constants/Routes';
import DateTimePickerModal from 'react-native-modal-datetime-picker';
import {ADD_VACCINE_CERTIFICATE} from '../../../../constants/ApiRoutes';
import moment from 'moment';
import {PostRequest} from '../../../../utils/apiRequests';
import {setVaccine} from '../../../../redux/actions/user';
import {KeyboardAwareScrollView} from 'react-native-keyboard-aware-scrollview';
import ModalBox from '../../../../components/ResultModal';
import {getData} from '../../../../utils/asyncRequests';
import {APP_LANG} from '../../../../constants/Keys';

/**
 * this functional component is to get vaccination certificate data.
 * After getting data from user, it will be stored to server db.
 * @param {object} param0 navigation and route props
 * @returns component
 */
const VaccineCertificate = ({route, navigation}) => {
  const dispatch = useDispatch();
  const [processing, setProcessing] = useState(false);
  const [show, setShow] = useState(false);
  const [disabled, setDisabled] = useState(true);

  const [vaccineName, setVaccineName] = useState();
  const [dosesRequired, setDosesRequired] = useState('');
  const [dosesCompleted, setDosesCompleted] = useState('');
  const [firstDoseDate, setfirstDoseDate] = useState();
  const [hospitalName, setHospitalName] = useState();
  const [showModal, setShowModal] = useState(false);
  const [lan, setLan] = useState('');

  const photo = route.params.photo;
  const token = useSelector(state => state.user.token);

  const {t} = useTranslation('en');

  useEffect(() => {
    getLang();
  }, []);

  /**
   * Get current app language
   */
  const getLang = async () => {
    let lang = await getData(APP_LANG);
    setLan(lang);
  };

  /**
   * Send Post Request for adding vaccine certificate to server db
   */
  const validateInput = async () => {
    if (
      vaccineName &&
      dosesRequired &&
      dosesCompleted &&
      firstDoseDate &&
      hospitalName
    ) {
      var formData = new FormData();
      formData.append('name', vaccineName);
      formData.append('requiredDoses', dosesRequired);
      formData.append('completedDoses', dosesCompleted);
      formData.append('date', firstDoseDate);
      formData.append('hospitalName', hospitalName);
      formData.append('photo', {
        uri: photo,
        type: 'image/jpeg',
        name: 'photo.jpg',
      });
      let data = {
        vaccineName: vaccineName,
        completeDate: firstDoseDate,
        hospitalName: hospitalName,
        dosesRequired: dosesRequired,
        dosesCompleted: dosesCompleted,
      };
      setProcessing(true);
      await PostRequest(
        ADD_VACCINE_CERTIFICATE,
        formData,
        'Token ' + token,
        'multipart/form-data',
      )
        .then(async response => {
          if (response.status === 201) {
            if (dosesRequired == dosesCompleted) {
              dispatch(setVaccine(data));
              navigation.navigate(SUCCESS_TEST);
            } else {
              setShowModal(true);
            }
          }
        })
        .catch(error => {
          showSnackBar('Check your connection or try again');
        })
        .finally(() => {
          setProcessing(false);
        });
    } else {
      showSnackBar(t('common.fillAll'));
    }
  };

  /**
   * Check input password for validation
   * @param {String} vaccineName vaccine name
   * @param {String} dosesRequired required doses
   * @param {String} dosesCompleted completed doses
   * @param {String} firstDoseDate first dose date
   * @param {String} hospitalName hospital/clinic name
   */
  const CheckInputs = (
    vaccineName,
    dosesRequired,
    dosesCompleted,
    hospitalName,
    firstDoseDate,
  ) => {
    if (
      vaccineName &&
      dosesRequired &&
      dosesCompleted &&
      firstDoseDate &&
      hospitalName
    ) {
      setDisabled(false);
    } else {
      setDisabled(true);
    }
  };

  /**
   * This method generalize using of Snackbar
   * @param {string} message message to show on snackbar
   */
  const showSnackBar = message => {
    Snackbar.show({
      text: message,
      backgroundColor: AppColor.BgColorSecondary,
      textColor: AppColor.PrimaryLight,
      duration: Snackbar.LENGTH_SHORT,
    });
  };

  /**
   * show hide doses incomplete modal
   */
  const onPressClose = () => {
    setShowModal(false);
    navigation.navigate(SUCCESS_TEST);
  };

  return (
    <SafeAreaView style={styles.safeAreaView}>
      <View style={styles.container}>
        {/* <StatusBar translucent hidden={true}></StatusBar> */}
        <Headerlight
          title={t('results.title')}
          color={AppColor.BgColorSecondary}
          navigation={navigation}
        />
        <ModalBox
          message={t('common.dosesNotCompleted')}
          onPressClose={onPressClose}
          visible={showModal}
        />

        <KeyboardAwareScrollView showsVerticalScrollIndicator={false}>
          <View style={{alignItems: 'center', justifyContent: 'center'}}>
            <Separator height={responsiveHeight(10)} />
            <View style={styles.imageView}>
              <Image source={{uri: photo}} style={styles.image} />
            </View>
            <Separator height={responsiveHeight(1)} />
            <View style={styles.inputView}>
              <TextInput
                testID={'passwordInput'}
                style={styles.input}
                placeholder={t('results.vaccineName')}
                placeholderTextColor={AppColor.TextColorLight}
                onChangeText={text => {
                  setVaccineName(text);
                  CheckInputs(
                    text,
                    dosesRequired,
                    dosesCompleted,
                    hospitalName,
                    firstDoseDate,
                  );
                }}
              />
            </View>
            <View
              style={[
                styles.bottomRow,
                {
                  height: responsiveHeight(4),
                  width: responsiveWidth(78),
                  marginRight: responsiveWidth(2),
                },
              ]}>
              <Text style={styles.rowText}>{t('results.reqDoses')}</Text>
              <View style={[styles.inputViewSmall]}>
                <TextInput
                  style={[styles.inputSmall]}
                  keyboardType={'decimal-pad'}
                  maxLength={2}
                  placeholder={dosesRequired ? dosesRequired : '0'}
                  value={dosesRequired}
                  placeholderTextColor={AppColor.TextColorLight}
                  onChangeText={text => {
                    setDosesRequired(text);
                    CheckInputs(
                      vaccineName,
                      text,
                      dosesCompleted,
                      hospitalName,
                      firstDoseDate,
                    );
                  }}
                />
                <TouchableOpacity
                  style={{
                    flex: 1,
                    alignItems: 'center',
                    justifyContent: 'center',
                  }}>
                  <DropDown
                    height={responsiveHeight(1.5)}
                    width={responsiveHeight(1.5)}
                    fill={AppColor.LogoColorLight}
                  />
                </TouchableOpacity>
              </View>
            </View>
            <View
              style={[
                styles.bottomRow,
                {
                  height: responsiveHeight(5),
                  width: responsiveWidth(78),
                  marginRight: responsiveWidth(2),
                },
              ]}>
              <Text style={styles.rowText}>{t('results.completeDoses')} </Text>
              <View style={[styles.inputViewSmall]}>
                <TextInput
                  placeholder={dosesCompleted ? dosesCompleted : '0'}
                  keyboardType={'decimal-pad'}
                  maxLength={2}
                  value={dosesCompleted}
                  placeholderTextColor={AppColor.TextColorLight}
                  onChangeText={text => {
                    setDosesCompleted(text);
                    CheckInputs(
                      vaccineName,
                      dosesRequired,
                      text,
                      hospitalName,
                      firstDoseDate,
                    );
                  }}
                  style={styles.inputSmall}
                />
                <TouchableOpacity
                  style={{
                    flex: 1,
                    alignItems: 'center',
                    justifyContent: 'center',
                  }}>
                  <DropDown
                    height={responsiveHeight(1.5)}
                    width={responsiveHeight(1.5)}
                    fill={AppColor.LogoColorLight}
                  />
                </TouchableOpacity>
              </View>
            </View>
            <Text style={styles.terms}>{t('results.qrNote')}</Text>
            <View
              style={[
                styles.bottomRow,
                {
                  height: responsiveHeight(5),
                  width: responsiveWidth(80),
                  marginLeft: responsiveWidth(6),
                  marginTop: responsiveHeight(1),
                },
              ]}>
              <View style={{flexDirection: 'row'}}>
                {lan == 'en' && (
                  <View style={{flexDirection: 'row'}}>
                    <Text style={styles.rowText}>1</Text>
                    <Text
                      style={{
                        fontSize: responsiveFontSize(1) * 1.1,
                        lineHeight: 26 * 0.5,
                        marginBottom: responsiveHeight(0.5),
                        color: AppColor.TextColorLight3,
                        textAlignVertical: 'top',
                      }}>
                      st
                    </Text>
                  </View>
                )}

                <Text style={styles.rowText}>{t('results.firstDose')}:</Text>
              </View>
              <TouchableOpacity
                onPress={() => setShow(!show)}
                style={[
                  styles.inputView,
                  {
                    height: responsiveHeight(6),
                    width: responsiveWidth(45),
                    // paddingTop: 15,
                  },
                ]}>
                <Text
                  // editable={false}
                  // value={
                  //   firstDoseDate
                  //     ? moment(firstDoseDate).format('YYYY/MM/DD')
                  //     : ''
                  // }
                  style={[
                    styles.input,
                    {
                      width: responsiveWidth(30),
                      marginRight: responsiveWidth(1),
                      paddingTop: 10,
                      color: firstDoseDate
                        ? AppColor.TextColorDark
                        : AppColor.TextColorLight3,
                    },
                  ]}
                  // placeholder={'yyyy/mm/dd'}
                  // placeholderTextColor={AppColor.TextColorLight}
                >
                  {firstDoseDate
                    ? moment(firstDoseDate).format('YYYY/MM/DD')
                    : 'yyyy/mm/dd'}
                </Text>
                <Cal
                  height={responsiveHeight(2.4)}
                  width={responsiveHeight(2.4)}
                  fill={AppColor.LogoColorLight}
                />
              </TouchableOpacity>
            </View>
            <Separator height={responsiveHeight(7)} />
            <View style={styles.inputView}>
              <TextInput
                testID={'passwordInput'}
                style={styles.input}
                placeholder={t('results.hospitalName')}
                placeholderTextColor={AppColor.TextColorLight}
                onChangeText={text => {
                  setHospitalName(text),
                    CheckInputs(
                      vaccineName,
                      dosesRequired,
                      dosesCompleted,
                      text,
                      firstDoseDate,
                    );
                }}
              />
            </View>
            <Separator height={responsiveHeight(1)} />
            <Button
              disabled={disabled}
              showLoading={processing}
              onPress={() => validateInput()}
              title={t('results.confirm')}
            />
            <DateTimePickerModal
              isVisible={show}
              date={new Date()}
              mode={'datetime'}
              maximumDate={new Date()}
              onConfirm={val => {
                let newdate = moment(val).format('YYYY-MM-DD HH:mm');
                setfirstDoseDate(newdate);
                setShow(!show);
                CheckInputs(
                  vaccineName,
                  dosesRequired,
                  dosesCompleted,
                  hospitalName,
                  newdate,
                );
              }}
              onCancel={() => setShow(false)}
            />
            <Separator height={responsiveHeight(4)} />
          </View>
        </KeyboardAwareScrollView>
      </View>
    </SafeAreaView>
  );
};

    src/screens/authenticated/siteInfoEdit/index.js

    import React, {useEffect, useState} from 'react';
import {SafeAreaView, StyleSheet, Text, View, TextInput} from 'react-native';
import {useTranslation} from 'react-i18next';
import CheckBox from '@react-native-community/checkbox';
import Icon from 'react-native-vector-icons/FontAwesome5';
import {
  responsiveWidth,
  responsiveHeight,
  responsiveFontSize,
} from 'react-native-responsive-dimensions';
import Button from '../../../components/Button';
import Separator from '../../../components/Separator';
import {AppColor} from '../../../constants/Colors';
import Headerlight from '../../../components/Headerlight';
import {UPDATE_SITE} from '../../../constants/ApiRoutes';
import Snackbar from 'react-native-snackbar';
import {PostRequest} from '../../../utils/apiRequests';
import {useDispatch, useSelector} from 'react-redux';
import {setRegisteredSite} from '../../../redux/actions/user';
import BottomSheet from '../../../components/BottomSheet';
import {TouchableOpacity} from 'react-native-gesture-handler';
import {KeyboardAwareScrollView} from 'react-native-keyboard-aware-scrollview';

const DATA2 = [12];

/**
 * this functional component is for editing site info.
 * @param {object} param0 navigation and route props
 * @returns component
 */
const SiteInfoEdit = ({route, navigation}) => {
  const {t} = useTranslation('en');

  const obj = route.params.data;
  const [vaccinated, setVaccinated] = useState(obj.vaccinated);
  const [siteId, setSiteId] = useState(obj._id);
  const [pcrNegative, setPcrNegative] = useState(obj.pcr);
  const [notTested, setNotTested] = useState(obj.notTested);
  const [name, setName] = useState(obj.name);
  const [address, setAddress] = useState(obj.address);
  const [phone, setPhone] = useState(obj.phone);
  const [date, setDate] = useState(obj.vaccinationTime);
  const [enDate, setEnDate] = useState(obj.vaccinationTime);
  const [time, setTime] = useState(obj.pcrTime);
  const [disabled, setDisabled] = useState(false);
  const [processing, setProcessing] = useState(false);
  const token = useSelector(state => state.user.token);
  const [modalVisible, setModalVisible] = useState(false);
  const [modalVisible2, setModalVisible2] = useState(false);

  global.monthsData = [
    {
      name: '1 ' + t('siteManager.month'),
      en: '1 Month',
    },
    {
      name: '2 ' + t('siteManager.month'),
      en: '2 Month',
    },
    {
      name: '3 ' + t('siteManager.month'),
      en: '3 Month',
    },
    {
      name: '4 ' + t('siteManager.month'),
      en: '4 Month',
    },
    {
      name: '5 ' + t('siteManager.month'),
      en: '5 Month',
    },
    {
      name: '6 ' + t('siteManager.month'),
      en: '6 Month',
    },
  ];
  const dispatch = useDispatch();

  useEffect(() => {
    for (let index = 0; index < 60; index++) {
      DATA2.push(DATA2[index] + 1);
    }
  }, []);

  /**
   * Send Post Request for registering site
   */
  const validateInput = async () => {
    let reqObj = [];

    let pattern = '\\d{2}\\d{9}';
    if (!phone.match(pattern) && phone) {
      showSnackBar(t('common.enterNumber'));
      return;
    }

    if (vaccinated) {
      if (date.trim()) {
        reqObj.push({type: 'vaccinated', time: enDate});
      } else {
        showSnackBar(t('common.addTime'));
        return;
      }
    }
    if (pcrNegative) {
      if (time.trim()) {
        reqObj.push({type: 'pcr', time: time.split(' ')[0] + ' Hours'});
      } else {
        showSnackBar(t('common.addPcrTime'));
        return;
      }
    }
    if (notTested) {
      reqObj.push({type: 'notTested'});
    }
    if (name && address && phone && (vaccinated || pcrNegative || notTested)) {
      let obj = {
        name: name,
        address: address,
        phone: phone,
        requirements: JSON.stringify(reqObj),
        siteId: siteId,
      };
      setProcessing(true);

      await PostRequest(UPDATE_SITE, obj, 'Token ' + token)
        .then(async response => {
          if (response.status === 200) {
            dispatch(setRegisteredSite(obj));
            showSnackBar(t('common.siteUpdated'));
            navigation.goBack();
          }
        })
        .catch(error => {
          if (error.response.status === 401) {
            if (
              error.response.data.message ===
              'phone with value ' +
                phone +
                ' fails to match the required pattern'
            ) {
              showSnackBar(t('common.invalidNumber'));
            }
            if (
              error.response.data.message ===
              "You have already registered a site, you can't register more sites"
            ) {
              showSnackBar(t('common.alredyRegister'));
            }
          }
          if (error.response.status === 409) {
            if (
              error.response.data.message ===
              "You have already registered a site, you can't register more sites"
            ) {
              showSnackBar(t('common.alredyRegister'));
            }
          } else {
            showSnackBar(t('common.checkConnection'));
          }
        })
        .finally(() => {
          setProcessing(false);
        });
    } else {
      showSnackBar(t('common.fillAll'));
    }
  };

  /**
   * This method generalize using of Snackbar
   * @param {string} message message to show on snackbar
   */
  const showSnackBar = message => {
    Snackbar.show({
      text: message,
      backgroundColor: AppColor.BgColorSecondary,
      textColor: AppColor.PrimaryLight,
      duration: Snackbar.LENGTH_SHORT,
    });
  };

  /**
   * hide modal with months
   */
  const onClose2 = () => {
    setModalVisible2(false);
  };

  /**
   * hide modal with hours
   */
  const onClose = item => {
    setModalVisible(false);
  };

  /**
   * Check input fields
   */
  const checkFields = (
    name,
    address,
    phone,
    vaccinated,
    pcrNegative,
    notTested,
  ) => {
    if (name && address && phone && (vaccinated || pcrNegative || notTested)) {
      setDisabled(false);
    } else {
      setDisabled(true);
    }
  };

  const onSelectVac = item => {
    setDate(item.name);
    setEnDate(item.en);
    setModalVisible(false);
    // checkFields();
  };
  const onSelectPcr = item => {
    const t1 = item.split(' ')[0];
    setTime(t1 + ' ' + t('siteManager.hours'));
    //  checkFields();
    setModalVisible2(false);
  };
  return (
    <SafeAreaView style={styles.safeAreaView}>
      <View style={styles.container}>
        <BottomSheet
          data={global.monthsData}
          visible={modalVisible}
          onClose={onClose}
          onSelect={onSelectVac}
        />
        <BottomSheet
          data={DATA2}
          visible={modalVisible2}
          onClose={onClose2}
          onSelect={onSelectPcr}
        />
        <Headerlight
          title={t('siteManager.title2')}
          color={AppColor.BtnSecondaryColor}
          navigation={navigation}
        />
        <KeyboardAwareScrollView showsVerticalScrollIndicator={false}>
          <Separator height={responsiveHeight(15)} />
          <View style={styles.inputView}>
            <TextInput
              style={styles.input}
              placeholder={t('siteManager.name')}
              onChangeText={text => {
                checkFields(
                  text,
                  address,
                  phone,
                  vaccinated,
                  pcrNegative,
                  notTested,
                );

                setName(text);
              }}
              value={name}
              placeholderTextColor={AppColor.TextColorLight}
            />
          </View>
          <View style={styles.inputView}>
            <TextInput
              style={styles.input}
              placeholder={t('siteManager.address')}
              onChangeText={text => {
                setAddress(text);
                checkFields(
                  name,
                  text,
                  phone,
                  vaccinated,
                  pcrNegative,
                  notTested,
                );
              }}
              value={address}
              placeholderTextColor={AppColor.TextColorLight}
            />
          </View>
          <View style={styles.inputView}>
            <TextInput
              keyboardType={'phone-pad'}
              style={styles.input}
              placeholder={t('siteManager.phone')}
              onChangeText={text => {
                setPhone(text),
                  checkFields(
                    name,
                    address,
                    text,
                    vaccinated,
                    pcrNegative,
                    notTested,
                  );
              }}
              value={phone}
              maxLength={11}
              placeholderTextColor={AppColor.TextColorLight}
            />
          </View>
          <Text style={styles.middleText}>{t('siteManager.note')}</Text>
          <View style={styles.bottomRow}>
            <View style={styles.innerItems}>
              <CheckBox
                disabled={false}
                tintColors={{
                  true: AppColor.Secondary,
                  false: AppColor.Secondary,
                }}
                onTintColor={AppColor.Secondary}
                onCheckColor={AppColor.Secondary}
                value={vaccinated}
                onValueChange={newValue => {
                  checkFields(
                    name,
                    address,
                    phone,
                    newValue,
                    pcrNegative,
                    notTested,
                  ),
                    setVaccinated(newValue);
                }}
              />

              <Text style={styles.rowText}>{t('siteManager.vaccinated')}</Text>
            </View>
            <TouchableOpacity
              onPress={() => setModalVisible(true)}
              style={styles.smallinputView}>
              <Text style={styles.smallinput}>
                {date.trim()
                  ? date.split(' ')[0] + ' ' + t('siteManager.month')
                  : t('siteManager.select')}
              </Text>
              <Icon
                name="calendar"
                size={responsiveFontSize(1.9)}
                color={AppColor.TextColorLight3}
              />
            </TouchableOpacity>
          </View>
          <View style={styles.bottomRow}>
            <View style={styles.innerItems}>
              <CheckBox
                disabled={false}
                tintColors={{
                  true: AppColor.Secondary,
                  false: AppColor.Secondary,
                }}
                onTintColor={AppColor.Secondary}
                onCheckColor={AppColor.Secondary}
                value={pcrNegative}
                onValueChange={newValue => {
                  checkFields(
                    name,
                    address,
                    phone,
                    vaccinated,
                    newValue,
                    notTested,
                  ),
                    setPcrNegative(newValue);
                }}
              />
              <Text style={styles.rowText}>{t('siteManager.pcrNegative')}</Text>
            </View>
            <TouchableOpacity
              onPress={() => setModalVisible2(true)}
              style={styles.smallinputView}>
              <Text style={styles.smallinput}>
                {time.trim()
                  ? time.split(' ')[0] + ' ' + t('siteManager.hours')
                  : t('siteManager.select')}
              </Text>
              <Icon
                name="clock"
                size={responsiveFontSize(1.9)}
                color={AppColor.TextColorLight3}
              />
            </TouchableOpacity>
          </View>
          <View style={styles.bottomRow}>
            <View style={styles.innerItems}>
              <CheckBox
                disabled={false}
                tintColors={{
                  true: AppColor.Secondary,
                  false: AppColor.Secondary,
                }}
                onTintColor={AppColor.Secondary}
                onCheckColor={AppColor.Secondary}
                value={notTested}
                onValueChange={newValue => {
                  setNotTested(newValue);
                  checkFields(
                    name,
                    address,
                    phone,
                    vaccinated,
                    pcrNegative,
                    newValue,
                  );
                }}
              />
              <Text style={styles.rowText}>{t('siteManager.notTested')}</Text>
            </View>
          </View>
          <Separator height={responsiveHeight(6)} />
          <Button
            showLoading={processing}
            disabled={disabled}
            onPress={() => validateInput()}
            title={t('siteManager.done')}
          />
          <Separator height={responsiveHeight(4)} />
        </KeyboardAwareScrollView>
      </View>
    </SafeAreaView>
  );
};

const styles = StyleSheet.create({
  safeAreaView: {
    flex: 1,
    backgroundColor: '#fff',
  },
  container: {
    flex: 1,

    backgroundColor: '#fff',
    alignItems: 'center',
    justifyContent: 'center',
  },

  inputView: {
    height: responsiveHeight(7),
    borderRadius: responsiveHeight(7),
    borderWidth: 1.5,
    borderColor: AppColor.BgColorSecondary,
    marginVertical: responsiveHeight(1),
    justifyContent: 'center',
    width: responsiveWidth(85),
    alignItems: 'center',
    flexDirection: 'row',
  },
  input: {
    width: responsiveWidth(75),
    alignSelf: 'center',
    height: responsiveHeight(5),
    color: AppColor.TextColor,
    padding: 0,
    fontSize: responsiveFontSize(2.1),
  },
  middleText: {
    fontSize: responsiveFontSize(2),
    marginTop: responsiveHeight(3),
    marginBottom: responsiveHeight(2),
    color: AppColor.TextColorLight3,
    marginLeft: responsiveWidth(5),
    width: responsiveWidth(80),
  },
  bottomRow: {
    flexDirection: 'row',
    width: responsiveWidth(81),
    height: responsiveHeight(6),
    marginLeft: responsiveWidth(4),
    justifyContent: 'space-between',
    alignItems: 'center',
  },
  innerItems: {
    flexDirection: 'row',
    width: responsiveWidth(27),
    alignItems: 'center',
  },
  rowText: {
    fontSize: responsiveFontSize(1.8),
    marginLeft: responsiveWidth(2),
    width: responsiveWidth(30),

  },
  smallinput: {
    width: responsiveWidth(15),
    marginRight: responsiveWidth(1),
    alignSelf: 'center',

    padding: 0,
    fontSize: responsiveFontSize(1.4),
  },
  smallinputView: {
    height: responsiveHeight(4.5),

    borderWidth: 1.5,
    borderColor: AppColor.BgColorSecondary,
    marginVertical: responsiveHeight(1),
    justifyContent: 'center',
    width: responsiveWidth(25),
    alignItems: 'center',
    flexDirection: 'row',
  },
});

export default SiteInfoEdit;
